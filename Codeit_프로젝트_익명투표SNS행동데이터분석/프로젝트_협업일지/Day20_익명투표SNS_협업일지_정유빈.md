# event_datetime_01에서 날짜만 추출 후 groupby로 바로 집계

df_dau = (
    df3_sorted
    .groupby(pd.to_datetime(df3_sorted['event_datetime_01']).dt.date)['user_id']
    .nunique()
    .reset_index()
    .rename(columns={'event_datetime_01': 'event_date', 'user_id': 'unique_user_count'})
)

df_dau
